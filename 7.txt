            let start = format(time);
            time.setMinutes(time.getMinutes() + 45);
            let end = format(time);

            const activityData = activityIcons[(i - 1) % activityIcons.length];

              type: "lesson",
              id: i,
              name: subjects[subjectIdx],
              topic: lessonTopics[subjectIdx],
              activityIcon: activityData.icon,
              activityLabel: activityData.label,
              time: `${start} - ${end}`,
              hasVideo: i % 2 !== 0,
              location: "القاعة " + (100 + i),
              content: "شرح مفصل حول " + lessonTopics[subjectIdx] + " مع تطبيقات عملية."

              schedule.push({
                type: "break",
                name: "فسحة مدرسية",
                time: `${bStart} - ${format(time)}`
              });
            }
          }
          return schedule;
